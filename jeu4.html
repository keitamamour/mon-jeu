import random import time score_joueur = 0 score_ordi = 0 while True: choix_ordi = random.choice(["pierre", "feuille", "ciseaux"]) choix_joueur = input("Choisis pierre, feuille ou ciseaux : ").lower() if choix_joueur not in ["pierre", "feuille", "ciseaux"]: print("Choix invalide !") continue print("\nPierre...") time.sleep(0.5) print("Feuille...") time.sleep(0.5) print("Ciseaux...") time.sleep(0.5) print("\n✊✋✌️ GO !\n") time.sleep(0.5) print("Toi :", choix_joueur) print("Ordinateur :", choix_ordi) if choix_joueur == choix_ordi: print("Égalité !") elif (choix_joueur == "pierre" and choix_ordi == "ciseaux") or \ (choix_joueur == "feuille" and choix_ordi == "pierre") or \ (choix_joueur == "ciseaux" and choix_ordi == "feuille"): print("Tu gagnes !") score_joueur += 1 else: print("Tu perds !") score_ordi += 1 print("Score -> Toi:", score_joueur, "| Ordi:", score_ordi) replay = input("\nRejouer ? (oui/non) : ").lower() if replay != "oui": break print("Merci d'avoir joué ! 👋")
